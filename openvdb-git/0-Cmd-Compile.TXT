3) MAKEPKG DANS TMP EN USER
====================================
!!! EN ROOT
# pacman -Syu

!!! EN USER

rm -f *~ \
&& rm -rf /tmp/build \
&& mkdir /tmp/build \
&& cd /tmp/build \
&& cp -v $HOME/05-builds/openvdb-git/PKGBUILD . \
&& cp -v $HOME/05-builds/openvdb-git/openvdb-git-3.2.0.20160810.tar.gz . \
&& LC_ALL=C makepkg -s \
&& cd $HOME/05-builds/openvdb-git \
&& cp -v /tmp/build/openvdb-git-*-x86_64.pkg.tar.* . \
